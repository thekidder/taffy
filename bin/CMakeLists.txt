CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

#needed to install SORE_DLL
FIND_PACKAGE(SORE REQUIRED)

ADD_SUBDIRECTORY(data)

INSTALL(FILES settings.ini DESTINATION . CONFIGURATIONS Release)
INSTALL(FILES settings_debug.ini DESTINATION . CONFIGURATIONS Debug RENAME settings.ini)
INSTALL(DIRECTORY logs DESTINATION . )

IF(WIN32)
#install debug and release SORE DLLs
INSTALL(FILES ${SORE_DLL_DEBUG} DESTINATION . CONFIGURATIONS Debug)
INSTALL(FILES ${SORE_DLL_RELEASE} DESTINATION . CONFIGURATIONS Release)
#install other required DLLs
INSTALL(FILES zlib1.dll SDL.dll glew32.dll freetype.dll DESTINATION .)
ELSE()
INSTALL(FILES ${SORE_LIBRARY_DEBUG} DESTINATION . CONFIGURATIONS Debug)
INSTALL(FILES ${SORE_LIBRARY_RELEASE} DESTINATION . CONFIGURATIONS Release)
ENDIF()
